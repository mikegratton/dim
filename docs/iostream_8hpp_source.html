22bool extract_formatted_quantity(formatted_quantity<Scalar>& o_formatted, std::istream& is)
23{
24 char* cursor = o_formatted.symbol();
25 is >> o_formatted.value();
26 while (is && detail::isseparator(is.peek())) {
27 is.ignore();
28 }
29 char* end = o_formatted.symbol() + kMaxSymbol;
30 detail::unit_string_scanner scanner;
31 while (is && scanner.accept(is.peek()) && cursor < end) {
32 is.get(*cursor++);
33 }
34 if (cursor < end) {
35 *cursor = '\0';
36 } else {
37 *(end - 1) = '\0';
38 }
39 return !(o_formatted.is_bad() || scanner.state() == detail::unit_parse_state::kError || is.fail());
