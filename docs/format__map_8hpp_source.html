382bool parse_quantity(Q& o_q, typename Q::scalar value, char const* unit_str,
383 input_format_map<typename Q::scalar, typename Q::system> const& unit_map = get_default_format<Q>())
384{
385 o_q = unit_map.template to_quantity<Q>(value, unit_str);
386 if (!o_q.is_bad()) {
387 return true;
388 }
389 auto dynamic_q = detail::parse_standard_rep<typename Q::scalar, typename Q::system>(unit_str);
