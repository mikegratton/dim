262template <class Scalar>
263std::from_chars_result from_chars(char const* i_start, char const* i_end, formatted_quantity<Scalar>& o_formatted)
264{
265 Scalar s;
266 o_formatted.symbol()[0] = '\0';
267
268 std::from_chars_result result = parse_scalar(s, i_start, i_end);
269 if (result.ptr == i_start) {
270 if (result.ec == std::errc{}) {
271 result.ec = std::errc::invalid_argument;
272 }
273 o_formatted = formatted_quantity<Scalar>::bad_format();
274 return result;
275 }
276 o_formatted.value(s);
277 if (detail::isseparator(*result.ptr)) {
278 ++result.ptr;
279 }
280 char* cursor = o_formatted.symbol();
281 char* symbolEnd = o_formatted.symbol() + kMaxSymbol;
282 detail::unit_string_scanner scanner;
283 while (cursor < symbolEnd && result.ptr < i_end && scanner.accept(*result.ptr)) {
284 *cursor++ = *result.ptr++;
285 }
286 // Ensure the string is nul terminated
287 if (cursor == symbolEnd) {
288 result.ec = std::errc::no_buffer_space;
289 o_formatted.symbol()[kMaxSymbol - 1] = '\0';
290 } else {
291 *cursor = '\0';
292 }
293 // Pack in scanner errors
294 if (scanner.state() == detail::unit_parse_state::kError) {
295 result.ec = std::errc::invalid_argument;
296 }
297 return result;
298}
