91  Q format(typename Q::scalar const& s, const char* symbol) const
92  {
93  auto it = input_symbol.find(formatter<Q>::index());
94  if (it != input_symbol.end()) {
95  // std::cout << "Found map for index " << formatter<Q>::index() << "\n";
96  format_map<Q> const& map = static_cast<map_container<Q> const*>(it->second.get())->map;
97  auto it = map.find(symbol);
98  if (it != map.end()) { return it->second.input(s); }
99  }
100  // std::cout << "dynamic parse instead of map for " << formatter<Q>::index() << "\n";
101  double scale;
102  if (detail::parse_unit_dynamic<typename Q::unit>(scale, symbol)) { return Q(s * scale); }
103  return Q::bad_quantity();

308 std::istream& operator>>(std::istream& is, Q& out_q)
309 {
310  typename Q::scalar value;
311  char unit_string[128];
312  is >> value >> std::skipws >> unit_string;
313  char* unit_start = unit_string[0] == '_' ? unit_string + 1 : unit_string;
314  for (char* cursor = unit_start; *cursor != 0; cursor++) {
315  if (*cursor == ',') {
316  *cursor = '\0';
317  is.putback(',');
318  break;
319  }
320  }
321  if (std::has_facet<quantity_facet>(is.getloc())) {
322  out_q = std::use_facet<quantity_facet>(is.getloc()).format<Q>(value, unit_start);
